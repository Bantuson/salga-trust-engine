**GSAP suppression helper** (module scope):
```typescript
async function suppressGsap(page: Page) {
  await page.waitForLoadState('domcontentloaded');
  await page.evaluate(() => {
    if ((window as any).gsap) {
      (window as any).gsap.globalTimeline.timeScale(100);
    }
  });
}
```
