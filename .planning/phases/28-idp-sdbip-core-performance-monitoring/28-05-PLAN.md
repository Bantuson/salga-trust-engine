```python
class EvidenceService:
    async def scan_and_upload(
        self,
        file_content: bytes,
        filename: str,
        content_type: str,
        actual_id: UUID,
        user: User,
        db: AsyncSession,
    ) -> EvidenceDocument:
        """Scan file with ClamAV, then upload to Supabase Storage."""

        scan_status = "clean"  # default if scanning disabled

        if settings.CLAMAV_ENABLED:
            try:
                import clamd  # clamav-client provides clamd module
                cd = clamd.ClamdNetworkSocket(
                    host=settings.CLAMAV_HOST,
                    port=settings.CLAMAV_PORT,
                    timeout=30,
                )
                result = cd.instream(io.BytesIO(file_content))
                # result is like: {'stream': ('OK', None)} or {'stream': ('FOUND', 'virus_name')}
                status_tuple = result.get("stream", ("ERROR", None))
                if status_tuple[0] == "FOUND":
                    raise HTTPException(422, f"File rejected by virus scanner: {status_tuple[1]}")
                elif status_tuple[0] != "OK":
                    scan_status = "scan_failed"
                    if settings.ENVIRONMENT == "production":
                        raise HTTPException(422, "Virus scan failed — upload blocked in production")
                    else:
                        logger.warning(f"ClamAV scan returned non-OK: {status_tuple}")
            except HTTPException:
                raise
            except Exception as exc:
                scan_status = "scan_failed"
                if settings.ENVIRONMENT == "production":
                    raise HTTPException(422, f"Virus scan unavailable — upload blocked in production: {exc}")
                else:
                    logger.warning(f"ClamAV unavailable (fail-open in dev): {exc}")

        # Generate unique filename
        unique_filename = f"{uuid4().hex}_{filename}"
        bucket = f"salga-evidence-{user.tenant_id}"
        storage_path = f"actuals/{actual_id}/{unique_filename}"

        # Upload to Supabase Storage (or skip if storage not available)
        try:
            storage_service = StorageService()
            if storage_service._storage_available:
                # Use Supabase storage upload
                storage_service._supabase.storage.from_(bucket).upload(
                    storage_path, file_content,
                    {"content-type": content_type}
                )
        except Exception as exc:
            logger.warning(f"Storage upload failed (continuing with DB record): {exc}")

        # Create DB record
        doc = EvidenceDocument(
            tenant_id=user.tenant_id,
            actual_id=actual_id,
            filename=unique_filename,
            original_filename=filename,
            content_type=content_type,
            file_size=len(file_content),
            storage_path=storage_path,
            scan_status=scan_status,
            uploaded_by=str(user.id),
            created_by=str(user.id),
        )
        db.add(doc)
        await db.commit()
        await db.refresh(doc)
        return doc

    async def list_evidence(self, actual_id: UUID, db: AsyncSession) -> list[EvidenceDocument]:
        result = await db.execute(
            select(EvidenceDocument).where(EvidenceDocument.actual_id == actual_id)
        )
        return list(result.scalars().all())

    async def get_signed_url(self, doc_id: UUID, db: AsyncSession) -> str:
        doc = await db.get(EvidenceDocument, doc_id)
        if not doc:
            raise HTTPException(404, "Evidence document not found")
        try:
            storage_service = StorageService()
            if storage_service._storage_available:
                bucket = f"salga-evidence-{doc.tenant_id}"
                return storage_service._supabase.storage.from_(bucket).create_signed_url(
                    doc.storage_path, 3600  # 1 hour expiry
                )
        except Exception:
            pass
        return f"/api/v1/sdbip/evidence/{doc_id}/download"  # fallback
```

```python
async def validate_actual(self, actual_id: UUID, user: User, db: AsyncSession) -> SDBIPActual:
    actual = await db.get(SDBIPActual, actual_id)
    if not actual:
        raise HTTPException(404, "Actual not found")
    if actual.is_validated:
        raise HTTPException(422, "Actual is already validated")
    actual.is_validated = True
    actual.validated_by = str(user.id)
    actual.validated_at = datetime.now(timezone.utc)
    actual.updated_by = str(user.id)
    await db.commit()
    await db.refresh(actual)
    return actual
```
