<!-- Backend endpoints for departments -->
GET /api/v1/departments — list departments for current municipality
  Response: [{ id, name, code, director_id, director_name, is_active, parent_department_id }]
POST /api/v1/departments — create department
  Body: { name: string, code: string }
PATCH /api/v1/departments/{id} — update department
DELETE /api/v1/departments/{id} — deactivate (soft delete)

<!-- Backend endpoints for role approvals -->
GET /api/v1/auth/tier1-approvals — list pending Tier 1 approval requests
POST /api/v1/auth/tier1-approvals/{id}/approve — approve
