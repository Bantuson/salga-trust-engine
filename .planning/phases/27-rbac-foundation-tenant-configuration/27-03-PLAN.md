```python
from dataclasses import dataclass, asdict
from uuid import UUID
from sqlalchemy import select, func
from sqlalchemy.ext.asyncio import AsyncSession
from fastapi import Depends, HTTPException, status

from src.api.deps import get_current_user, get_db
from src.models.department import Department
from src.models.municipality import Municipality
from src.models.role_assignment import UserRoleAssignment
from src.models.user import User, UserRole


@dataclass
class PmsReadinessStatus:
    is_ready: bool
    municipality_configured: bool
    all_departments_have_directors: bool
    pms_officer_assigned: bool
    department_count: int
    departments_with_directors: int
    missing_directors: list[str]  # department names with no director


async def check_pms_readiness(municipality_id: UUID, tenant_id: str, db: AsyncSession) -> PmsReadinessStatus:
    """Compute PMS readiness checklist for a municipality."""

    # 1. Check municipality settings configured
    municipality = await db.get(Municipality, municipality_id)
    municipality_configured = (
        municipality is not None
        and municipality.settings_locked
    )

    # 2. Check all departments have assigned directors
    # Query departments for this tenant
    dept_result = await db.execute(
        select(Department)
        .where(Department.tenant_id == tenant_id)
        .where(Department.is_active == True)
    )
    departments = dept_result.scalars().all()
    department_count = len(departments)

    missing_directors = [d.name for d in departments if d.assigned_director_id is None]
    departments_with_directors = department_count - len(missing_directors)
    all_departments_have_directors = len(missing_directors) == 0 and department_count > 0

    # 3. Check at least 1 PMS officer assigned
    pms_result = await db.execute(
        select(func.count())
        .select_from(UserRoleAssignment)
        .where(UserRoleAssignment.tenant_id == tenant_id)
        .where(UserRoleAssignment.role == UserRole.PMS_OFFICER)
        .where(UserRoleAssignment.is_active == True)
    )
    pms_officer_count = pms_result.scalar_one()
    pms_officer_assigned = pms_officer_count > 0

    is_ready = municipality_configured and all_departments_have_directors and pms_officer_assigned

    return PmsReadinessStatus(
        is_ready=is_ready,
        municipality_configured=municipality_configured,
        all_departments_have_directors=all_departments_have_directors,
        pms_officer_assigned=pms_officer_assigned,
        department_count=department_count,
        departments_with_directors=departments_with_directors,
        missing_directors=missing_directors,
    )
```

Add `require_pms_ready()` dependency factory:
```python
def require_pms_ready():
    """FastAPI dependency that returns 403 + structured checklist if PMS not configured."""
    async def pms_gate(
        current_user: User = Depends(get_current_user),
        db: AsyncSession = Depends(get_db),
    ) -> User:
        readiness = await check_pms_readiness(
            current_user.municipality_id, current_user.tenant_id, db
        )
        if not readiness.is_ready:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail={
                    "code": "PMS_NOT_READY",
                    "message": "PMS configuration incomplete",
                    "checklist": asdict(readiness),
                }
            )
        return current_user
    return pms_gate
```

```tsx
import Tree from 'react-d3-tree';
import { useCallback, useState } from 'react';

interface OrgNode {
  name: string;
  attributes?: Record<string, string>;
  children?: OrgNode[];
}

interface OrganogramTreeProps {
  data: OrgNode;
  onNodeClick?: (node: OrgNode) => void;
}

export function OrganogramTree({ data, onNodeClick }: OrganogramTreeProps) {
  const [translate, setTranslate] = useState({ x: 0, y: 0 });
  const containerRef = useCallback((containerElem: HTMLDivElement | null) => {
    if (containerElem) {
      const { width } = containerElem.getBoundingClientRect();
      setTranslate({ x: width / 2, y: 50 });
    }
  }, []);

  return (
    <div ref={containerRef} style={{ width: '100%', height: '600px' }}>
      <Tree
        data={data}
        translate={translate}
        orientation="vertical"
        pathFunc="step"
        collapsible={true}
        initialDepth={2}
        separation={{ siblings: 1.5, nonSiblings: 2 }}
        renderCustomNodeElement={({ nodeDatum, toggleNode }) => (
          <g onClick={() => { toggleNode(); onNodeClick?.(nodeDatum as OrgNode); }}>
            <circle r={24} fill="#0D9488" stroke="#065F46" strokeWidth={2} />
            <text dy=".35em" textAnchor="middle" fill="white" fontSize={11} fontWeight={600}>
              {(nodeDatum.name as string).slice(0, 3).toUpperCase()}
            </text>
            <text dy="3em" textAnchor="middle" fill="#1F2937" fontSize={12} fontWeight={500}>
              {nodeDatum.name}
            </text>
            {nodeDatum.attributes?.director && (
              <text dy="4.5em" textAnchor="middle" fill="#6B7280" fontSize={10}>
                {nodeDatum.attributes.director}
              </text>
            )}
          </g>
        )}
      />
    </div>
  );
}
```
